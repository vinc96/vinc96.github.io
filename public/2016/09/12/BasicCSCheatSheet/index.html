刪除YouTube播放列表視頻代碼（在瀏覽器console內輸入）：

function removeVideos(){
var el = document.getElementsByClassName(‘pl-video-edit-remove’);
var i=0;
setInterval(function(){el[i++].click()}, 50);
}
removeVideos();
window.onerror = function(){removeVideos();}
